# Specify our source files
set(ShaderSupport_SOURCES
    "ShaderCode.cpp"
)
set(ShaderSupport_HEADERS
    "ShaderSupport.pkg"
    "IShaderCode.h"
    "IShaderCode.inl"
    "ShaderCode.h"
)

# Define our project targets
add_library(ShaderSupport SHARED ${ShaderSupport_SOURCES} ${ShaderSupport_HEADERS} ${COBALT_VERSION_FILES})
add_library(Cobalt::ShaderSupport ALIAS ShaderSupport)
register_clang_format_and_tidy_target(ShaderSupport)
cobalt_apply_compiler_warning_settings(ShaderSupport)
target_include_directories(ShaderSupport PUBLIC ${COBALT_INTERNAL_INCLUDE_DIR})

# Set custom compilation/linker settings
target_compile_definitions(ShaderSupport PRIVATE SHADERSUPPORT_DLL)

# Specify our project dependencies
target_link_libraries(ShaderSupport
    PRIVATE
        Cobalt::RendererInterface
        Cobalt::LoggingInterface
        spirv-crossBundle
        SPIRV-ToolsBundle
        glslangBundle
        Vulkan-Headers
)

# Transfer required files to their output locations
cobalt_configure_sdk_install(ShaderSupport OMIT_LIB)

# Define our virtual project folder and file structure
set_target_properties(ShaderSupport PROPERTIES FOLDER "Libraries/Private")
source_group(TREE "${CMAKE_CURRENT_SOURCE_DIR}" FILES ${ShaderSupport_SOURCES} ${ShaderSupport_HEADERS})
make_paths_relative(COBALT_VERSION_FILES_RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} ${COBALT_VERSION_FILES})
source_group("Autogenerated" FILES ${COBALT_VERSION_FILES_RELATIVE})
