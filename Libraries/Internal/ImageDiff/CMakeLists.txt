# Specify our source files
set(ImageDiff_SOURCES
    "ImageRGBA.cpp"
    "ImageDiff.cpp"
)
set(ImageDiff_HEADERS
    "ImageDiff.pkg"
    "IImageDiff.h"
    "IImageDiff.inl"
    "IImageRGBA.h"
    "IImageRGBA.inl"
    "ImageRGBA.h"
    "ImageDiff.h"
)

# Define our project targets
add_library(ImageDiff SHARED ${ImageDiff_SOURCES} ${ImageDiff_HEADERS} ${COBALT_VERSION_FILES})
add_library(Cobalt::ImageDiff ALIAS ImageDiff)
register_clang_format_and_tidy_target(ImageDiff)
cobalt_apply_compiler_warning_settings(ImageDiff)
target_include_directories(ImageDiff PUBLIC ${COBALT_INTERNAL_INCLUDE_DIR})

# Set custom compilation/linker settings
target_compile_definitions(ImageDiff PRIVATE IMAGEDIFF_DLL)

# Specify our project dependencies
target_link_libraries(ImageDiff
    PRIVATE
        Cobalt::LoggingInterface
        Cobalt::Marshalling
        png_static
        zlibstatic
)

# Define our virtual project folder and file structure
set_target_properties(ImageDiff PROPERTIES FOLDER "Libraries/Private")
source_group(TREE "${CMAKE_CURRENT_SOURCE_DIR}" FILES ${ImageDiff_SOURCES} ${ImageDiff_HEADERS})
make_paths_relative(COBALT_VERSION_FILES_RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} ${COBALT_VERSION_FILES})
source_group("Autogenerated" FILES ${COBALT_VERSION_FILES_RELATIVE})
