# Specify our source files
set(ShaderConverter_SOURCES
    "main.cpp"
)

# Override some files as objective C++ for Cocoa support on macos
if(APPLE)
    set_source_files_properties(main.cpp PROPERTIES LANGUAGE OBJCXX)
endif()

# Define our project targets
add_executable(ShaderConverter ${ShaderConverter_SOURCES} ${COBALT_VERSION_FILES})
register_clang_format_and_tidy_target(ShaderConverter)
cobalt_apply_compiler_warning_settings(ShaderConverter)
add_rpath(ShaderConverter ".")

# Specify our project dependencies
target_link_libraries(ShaderConverter
    PRIVATE
        Cobalt::CrashHandling
        Cobalt::ShaderSupport
)
if(APPLE)
    target_link_libraries(ShaderConverter PRIVATE "-framework Foundation")
endif()

# Transfer required files to their output locations
cobalt_stage_deps(ShaderConverter "")
cobalt_install_with_deps(ShaderConverter "Utilities/ShaderConverter/${COBALT_ARCH}")

# Define our virtual project folder and file structure
set_target_properties(ShaderConverter PROPERTIES FOLDER "Utilities")
source_group(TREE "${CMAKE_CURRENT_SOURCE_DIR}" FILES ${ShaderConverter_SOURCES} ${ShaderConverter_HEADERS} ${ShaderConverter_RESOURCES})
make_paths_relative(COBALT_VERSION_FILES_RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} ${COBALT_VERSION_FILES})
source_group("Autogenerated" FILES ${COBALT_VERSION_FILES_RELATIVE})
