# Specify our source files
set(CBindingsTest_SOURCES
    "main.c"
)

# Define our project targets
add_executable(CBindingsTest ${CBindingsTest_SOURCES} ${COBALT_VERSION_FILES})
register_clang_format_and_tidy_target(CBindingsTest)
cobalt_apply_compiler_warning_settings(CBindingsTest)
add_rpath(CBindingsTest ".")

# Set custom compilation/linker settings
if(WIN32)
    target_compile_definitions(CBindingsTest PRIVATE _CRT_SECURE_NO_WARNINGS)
endif()

# Specify our project dependencies
target_link_libraries(CBindingsTest
    PRIVATE
        Cobalt::CobaltCBindings
        png_static
)
add_dependencies(
    CBindingsTest
    Cobalt::VulkanRenderer
)

# Transfer required files to their output locations
set(TARGET_INSTALL_DIR "${COBALT_OUTPUT_ROOT}/CBindingsTest_${COBALT_ARCH}_$<CONFIG>")
cobalt_install_with_deps(CBindingsTest "${COBALT_OUTPUT_ROOT}/CBindingsTest_${COBALT_ARCH}_$<CONFIG>")
cobalt_install_generated_bin("${COBALT_SDK_BIN_DIR}/" "${TARGET_INSTALL_DIR}")

# Define our virtual project folder and file structure
set_target_properties(CBindingsTest PROPERTIES FOLDER "Testing")
source_group(TREE "${CMAKE_CURRENT_SOURCE_DIR}" FILES ${CBindingsStaticTest_SOURCES})
make_paths_relative(COBALT_VERSION_FILES_RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} ${COBALT_VERSION_FILES})
source_group("Autogenerated" FILES ${COBALT_VERSION_FILES_RELATIVE})
